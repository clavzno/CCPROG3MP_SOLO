/**
 * This class is for console design purposes.
 */
public class Misc {

    /**
     * Clears the console screen.
     */
    public void clearScreen() {
        System.out.print("\033[H\033[2J");
        System.out.flush();
    }

    /**
     * Pauses the execution of the program for 2 seconds.
     * If there's no try/catch block, the program won't compile.
     */
    public void pauseScreen2Sec() {
        try {
            Thread.sleep(2000);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}
